# Phase 3 — Fix `ForgotPasswordView.swift` + `AuthManager.resetPassword()`

> **Depends on:** Phase 1 complete and building.

---

## The problem

`AuthManager.resetPassword()` previously wrote `newPassword` to
`staff_members.password`. That column is gone. The user also has no
Supabase Auth session during a password reset, so `supabase.auth.update()`
is unavailable client-side.

## The solution

A new `reset-password` edge function (`verify_jwt: false`) accepts a
short-lived server-side token + new password and calls
`auth.admin.updateUserById()`. The token is generated by `requestPasswordReset()`
and stored in a `password_reset_tokens` table (created in Phase 4).

---

## Step A — Create `supabase/functions/reset-password/index.ts`

Create this file in the repo. Phase 4 handles deployment to Supabase.

```typescript
// reset-password edge function
// verify_jwt: false — caller has no session during password reset.
// Validates a short-lived UUID token from password_reset_tokens table,
// then calls auth.admin.updateUserById() with the new password.

import { serve } from "https://deno.land/std@0.177.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

const SUPABASE_URL              = Deno.env.get("SUPABASE_URL")              ?? "";
const SUPABASE_SERVICE_ROLE_KEY = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY") ?? "";

const corsHeaders = {
  "Access-Control-Allow-Origin":  "*",
  "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type",
};

serve(async (req: Request) => {
  if (req.method === "OPTIONS") {
    return new Response(null, { headers: corsHeaders });
  }
  try {
    const { email, reset_token, new_password } = await req.json();

    if (!email || !reset_token || !new_password) {
      return new Response(JSON.stringify({ error: "Missing fields" }), {
        status: 400, headers: { "Content-Type": "application/json", ...corsHeaders },
      });
    }

    const admin = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY, {
      auth: { autoRefreshToken: false, persistSession: false },
    });

    // Validate the reset token
    const { data: tokens, error: tokenErr } = await admin
      .from("password_reset_tokens")
      .select("user_id, expires_at, used")
      .eq("email", email)
      .eq("token", reset_token)
      .limit(1);

    if (tokenErr || !tokens || tokens.length === 0) {
      return new Response(JSON.stringify({ error: "Invalid or expired token" }), {
        status: 401, headers: { "Content-Type": "application/json", ...corsHeaders },
      });
    }

    const tokenRow = tokens[0];

    if (tokenRow.used) {
      return new Response(JSON.stringify({ error: "Token already used" }), {
        status: 401, headers: { "Content-Type": "application/json", ...corsHeaders },
      });
    }

    if (new Date(tokenRow.expires_at) < new Date()) {
      return new Response(JSON.stringify({ error: "Token expired" }), {
        status: 401, headers: { "Content-Type": "application/json", ...corsHeaders },
      });
    }

    // Update Supabase Auth password (bcrypt-hashed server-side)
    const { error: updateErr } = await admin.auth.admin.updateUserById(
      tokenRow.user_id,
      { password: new_password }
    );

    if (updateErr) {
      return new Response(JSON.stringify({ error: updateErr.message }), {
        status: 500, headers: { "Content-Type": "application/json", ...corsHeaders },
      });
    }

    // Mark token as used (single-use enforcement)
    await admin
      .from("password_reset_tokens")
      .update({ used: true })
      .eq("token", reset_token);

    return new Response(JSON.stringify({ success: true }), {
      status: 200, headers: { "Content-Type": "application/json", ...corsHeaders },
    });
  } catch (err) {
    console.error("[reset-password] error:", err);
    return new Response(JSON.stringify({ error: String(err) }), {
      status: 500, headers: { "Content-Type": "application/json", ...corsHeaders },
    });
  }
});
```
